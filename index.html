def are_anagrams(str1, str2):
    return sorted(str1) == sorted(str2)

print(are_anagrams("listen", "silent"))
print(are_anagrams("hello", "world"))

prime.py


def is_prime(n):
    if n <= 1:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True
print(is_prime(7))
print(is_prime(11))

      from collections import Counter

  def most_frequent_number(lst):
      if not lst:
          return None
      count = Counter(lst)
      return count.most_common(1)[0][0]

  numbers = [1, 3, 1, 3, 2, 1]
  print(most_frequent_number(numbers))

        return None

                if name:
                    user["name"] = name
                if email:

                return True
        return False

        for user in users:
            if user["id"] == user_id:
                users.remove(user)
                return True
        return False

import re

def is_valid_email(email):
    pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
    return re.match(pattern, email) is not None

print(is_valid_email("user@domain.com"))   
print(is_valid_email("user@domain"))       
print(is_valid_email("user@.com"))

credentials.py


import hashlib

user_db = {}

def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()

def register(username, password):
    if username in user_db:
        return "User already exists"
    hashed = hash_password(password)
    user_db[username] = hashed
    return "Created new user"

def login(username, password):
    
    hashed = hash_password(password)
    if username in user_db and user_db[username] == hashed:
        return "Login Successful"
    return "Invalid credentials"

print(register("john", "mypassword"))  
print(login("john", "mypassword"))     
print(login("john", "wrongpass"))
